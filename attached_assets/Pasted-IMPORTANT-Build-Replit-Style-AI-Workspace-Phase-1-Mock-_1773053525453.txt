2. Prisma Schema (Core Entities)

model User {
  id            String    @id @default(cuid())
  email         String?   @unique
  projects      Project[]
}

model Project {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  specJson    Json?
  chats       Chat[]
  jobs        Job[]
}

model Message {
  id        String   @id @default(cuid())
  mode      String   @default("Discuss")
  content   String
  chatId    String
}

model Job {
  id        String   @id @default(cuid())
  status    String   @default("PENDING")
  logs      JobLog[]
}

model JobLog {
  id        String   @id @default(cuid())
  jobId     String
  job       Job      @relation(fields: [jobId], references: [id])
  level     String
  message   String
  createdAt DateTime @default(now())
}
